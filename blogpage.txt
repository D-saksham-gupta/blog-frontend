import { useState } from 'react'
import './BlogPage.css'

const sampleBlog = {
  id: 1,
  title: 'The Art of Mindful Writing: Finding Your Authentic Voice',
  description: 'Discover how to tap into your unique perspective and write with purpose. This comprehensive guide walks you through the journey of finding your authentic voice in a world full of noise.',
  image: 'https://images.pexels.com/photos/3636136/pexels-photo-3636136.jpeg?auto=compress&cs=tinysrgb&w=800',
  category: 'Writing Tips',
  author: 'Sarah Chen',
  authorImage: 'https://images.pexels.com/photos/1239291/pexels-photo-1239291.jpeg?auto=compress&cs=tinysrgb&w=400',
  date: 'November 10, 2024',
  likes: 2847,
  comments: [
    {
      id: 1,
      author: 'Emily Rodriguez',
      authorImage: 'https://images.pexels.com/photos/1181690/pexels-photo-1181690.jpeg?auto=compress&cs=tinysrgb&w=400',
      text: 'This article completely changed my perspective on writing. The section about finding your voice was incredibly insightful!',
      date: '2 days ago',
      likes: 124,
    },
    {
      id: 2,
      author: 'James Wilson',
      authorImage: 'https://images.pexels.com/photos/1222271/pexels-photo-1222271.jpeg?auto=compress&cs=tinysrgb&w=400',
      text: 'Great practical tips. I especially loved the exercises at the end of each section.',
      date: '1 day ago',
      likes: 89,
    },
  ],
}

function BlogPage() {
  const [isLiked, setIsLiked] = useState(false)
  const [likeCount, setLikeCount] = useState(sampleBlog.likes)
  const [comments, setComments] = useState(sampleBlog.comments)
  const [newComment, setNewComment] = useState('')
  const [showCommentForm, setShowCommentForm] = useState(false)

  const handleLike = () => {
    setIsLiked(!isLiked)
    setLikeCount(isLiked ? likeCount - 1 : likeCount + 1)
  }

  const handleCommentSubmit = (e) => {
    e.preventDefault()
    if (newComment.trim()) {
      const comment = {
        id: comments.length + 1,
        author: 'You',
        authorImage: 'https://images.pexels.com/photos/1239291/pexels-photo-1239291.jpeg?auto=compress&cs=tinysrgb&w=400',
        text: newComment,
        date: 'just now',
        likes: 0,
      }
      setComments([comment, ...comments])
      setNewComment('')
      setShowCommentForm(false)
    }
  }

  return (
    <div className="blog-page">
      <article className="blog-article">
        <div className="blog-header">
          <div className="blog-category">{sampleBlog.category}</div>
          <h1 className="blog-title">{sampleBlog.title}</h1>
          <div className="blog-meta">
            <div className="author-info">
              <img src={sampleBlog.authorImage} alt={sampleBlog.author} className="author-avatar" />
              <div>
                <p className="author-name">{sampleBlog.author}</p>
                <p className="blog-date">{sampleBlog.date}</p>
              </div>
            </div>
            <div className="blog-stats">
              <span className="stat">{likeCount} likes</span>
              <span className="stat">{comments.length} comments</span>
            </div>
          </div>
        </div>

        <img src={sampleBlog.image} alt={sampleBlog.title} className="blog-hero-image" />

        <div className="blog-content">
          <p className="blog-intro">{sampleBlog.description}</p>

          <h2>The Journey Begins</h2>
          <p>
            Writing is not just about putting words on a page. It's about connecting with your readers on a deeper level, sharing your unique perspective, and creating content that resonates with their hearts and minds. Many writers struggle with finding their authentic voice because they're too focused on what they think their audience wants to hear.
          </p>

          <h2>Understanding Your Authentic Voice</h2>
          <p>
            Your authentic voice is the unique combination of your experiences, values, perspectives, and personality that shines through in your writing. It's what makes your work distinctly yours. When you write with your authentic voice, your words carry more power and your readers can sense the genuine intention behind them.
          </p>
          <p>
            The key is to stop trying to sound like someone else. Instead, embrace your quirks, your perspectives, and your unique way of seeing the world. Your readers will appreciate the honesty and vulnerability you bring to your writing.
          </p>

          <h2>Practical Steps to Find Your Voice</h2>
          <p>
            Start by reading widely across different genres and authors. Pay attention to what resonates with you and why. Notice the patterns in the writing that speaks to you. Then, begin experimenting with your own writing style. Don't be afraid to take risks and try new approaches.
          </p>
          <p>
            Write regularly, even if it's just for yourself. The more you write, the more your natural voice will emerge. Over time, you'll develop a distinctive style that's uniquely yours.
          </p>

          <div className="blog-actions">
            <button
              className={`like-button ${isLiked ? 'liked' : ''}`}
              onClick={handleLike}
            >
              <span className="heart-icon">♥</span>
              <span>{isLiked ? 'Liked' : 'Like'}</span>
            </button>
            <button
              className="comment-button"
              onClick={() => setShowCommentForm(!showCommentForm)}
            >
              <span>Comment</span>
            </button>
            <button className="share-button">
              <span>Share</span>
            </button>
          </div>
        </div>
      </article>

      <aside className="blog-sidebar">
        <div className="author-card">
          <img src={sampleBlog.authorImage} alt={sampleBlog.author} className="author-card-image" />
          <h3 className="author-card-name">{sampleBlog.author}</h3>
          <p className="author-card-bio">
            A passionate writer and storyteller dedicated to helping others find their authentic voice through the power of words.
          </p>
          <button className="follow-button">Follow</button>
        </div>

        <div className="related-posts">
          <h3 className="related-posts-title">Related Articles</h3>
          <ul className="related-posts-list">
            <li><a href="#related">The Power of Storytelling in Modern Writing</a></li>
            <li><a href="#related">How to Overcome Writer's Block</a></li>
            <li><a href="#related">Building Your Reader Community</a></li>
          </ul>
        </div>
      </aside>

      <section className="comments-section">
        <div className="comments-container">
          <h2 className="comments-title">Comments ({comments.length})</h2>

          {showCommentForm && (
            <form className="comment-form" onSubmit={handleCommentSubmit}>
              <div className="comment-form-header">
                <img
                  src="https://images.pexels.com/photos/1239291/pexels-photo-1239291.jpeg?auto=compress&cs=tinysrgb&w=400"
                  alt="You"
                  className="comment-avatar"
                />
                <div className="comment-form-inputs">
                  <textarea
                    className="comment-textarea"
                    placeholder="Share your thoughts..."
                    value={newComment}
                    onChange={(e) => setNewComment(e.target.value)}
                    rows="3"
                  />
                  <div className="comment-form-actions">
                    <button type="submit" className="comment-submit-btn">Post Comment</button>
                    <button
                      type="button"
                      className="comment-cancel-btn"
                      onClick={() => setShowCommentForm(false)}
                    >
                      Cancel
                    </button>
                  </div>
                </div>
              </div>
            </form>
          )}

          {!showCommentForm && (
            <button
              className="add-comment-btn"
              onClick={() => setShowCommentForm(true)}
            >
              Add a comment...
            </button>
          )}

          <div className="comments-list">
            {comments.map((comment) => (
              <div key={comment.id} className="comment">
                <img src={comment.authorImage} alt={comment.author} className="comment-avatar" />
                <div className="comment-content">
                  <div className="comment-header">
                    <h4 className="comment-author">{comment.author}</h4>
                    <span className="comment-date">{comment.date}</span>
                  </div>
                  <p className="comment-text">{comment.text}</p>
                  <div className="comment-actions">
                    <button className="comment-like-btn">♥ {comment.likes} Like</button>
                    <button className="comment-reply-btn">Reply</button>
                  </div>
                </div>
              </div>
            ))}
          </div>
        </div>
      </section>
    </div>
  )
}

export default BlogPage
